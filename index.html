Backend / Python
scheduler_core.py 
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from .engine import KobeScheduler, JAPAN_CHILDCARE_LAW

router = APIRouter()

@router.post("/api/schedule/optimize")
async def optimize(payload: SchedulePayload, db: Session = Depends(get_db)):
    """保育士配置の最適化エンジン — 法定基準を自動充足"""
    engine = KobeScheduler(rules=JAPAN_CHILDCARE_LAW)
    result = await engine.solve(payload, db)
    return {"status": "ok", "schedule": result}
